type execState =
|Success of string * string
|Failure of string
type process_state = {
outbuf : Buffer.t;
errbuf : Buffer.t;
fd_out : Unix.file_descr;
fd_err : Unix.file_descr;
pid : int;
mutable out_closed : bool;
mutable err_closed : bool;
}
type spawn = unit -> process_state
val spawn : string list -> process_state
val process_processes : ('a * process_state) list -> ('a * process_state) * ('a * process_state) list
val terminate_process : 'a * process_state -> execState
val run_with_outputs : string list -> execState
type 'a schedule_op =
|Terminate
|WaitingTask
|AddProcess of 'a * process_state
|AddTask of 'a * spawn list list
|Retry
val schedule_op_to_string : 'a schedule_op -> string
type 'a schedule_task_group = {
mutable group_completion : int;
mutable group_next : ('a * spawn) list list;
}
type schedule_stats = {
mutable max_runqueue : int;
mutable nb_processes : int;
}
val schedule : int ->
(unit -> 'a schedule_op) ->
('a * process_state -> bool -> unit) -> schedule_stats
