type flavor =
|FlavorOffset
|FlavorPosition
type where =
|WhereStart
|WhereEnd
type subject =
|Left
|RightDollar of int
|RightNamed of string
type keyword =
|Dollar of int
|Position of subject * where * flavor
|PreviousError
|SyntaxError
val posvar : subject -> where -> flavor -> string
module KeywordSet : sig .. end with type elt = keyword
include Set.S
val from_list : elt list -> t
