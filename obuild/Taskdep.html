type taskdep_direction =
|FromChildren
|FromParent
type 'a taskdep = {
dag : 'a Dag.t;
nbSteps : int;
stepDone : ('a, unit) Hashtbl.t;
direction : taskdep_direction;
mutable curStep : int;
mutable nextTasks : 'a list;
}
val init_with : 'a Dag.t -> taskdep_direction -> 'a list -> 'a taskdep
val init_with_direction : 'a Dag.t -> taskdep_direction -> 'a taskdep
val init : 'a Dag.t -> 'a taskdep
val getnext_index : 'a taskdep -> int
val getnext : 'a taskdep -> (int * 'a) option
val getnext_choice : ('a list -> 'a) -> 'a taskdep -> (int * 'a) option
val markDone : 'a taskdep -> 'a -> unit
val isComplete : 'a taskdep -> bool
val linearize : 'a Dag.t -> taskdep_direction -> 'a list -> 'a list
val dump : ('a -> string) -> 'a taskdep -> unit
