Wiki management: entries, ATOM feeds, etc.
type body =
|File of string
|Html of Cow.Html.t
type entry = {
updated : Date.date;
author : Cow.Atom.author;
subject : string;
body : body;
permalink : string;
}
val html_of_author : Cow.Atom.author -> Cow.Xml.t
val atom_date : Date.date -> int * Date.month * int * int * int
val short_html_of_date : Date.date -> Cow.Xml.t
val body_of_entry : (string -> Cow.Html.t Lwt.t) -> entry -> Cow.Html.t Lwt.t
val compare_dates : entry -> entry -> int
val html_of_entry : ?want_date:bool -> (string -> Cow.Html.t Lwt.t) -> entry -> Cow.Xml.t Lwt.t
val html_of_index : (string -> ('a Cow.Xml.frag as 'a) list Lwt.t) -> Cow.Xml.t Lwt.t
val permalink : entry -> string
val html_of_recent_updates : entry list -> Cow.Xml.t
val html_of_page : ?disqus:string ->
content:('a Cow.Xml.frag as 'a) list Lwt.t ->
sidebar:('b Cow.Xml.frag as 'b) list -> Cow.Xml.t Lwt.t
